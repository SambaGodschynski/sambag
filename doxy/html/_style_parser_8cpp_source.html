00035                 color.setValues(c.getR(), c.getG(), c.getB(), c.getA());

00038         if ( !boost::regex_match(str, colorval) ) return;

00046         Number g = hex2Int(expr.substr(3,2)) / 255.0;
00047         Number b = hex2Int(expr.substr(5,2)) / 255.0;
00048         Number a = 255;
