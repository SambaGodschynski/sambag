00046         pI.push_back( std::make_pair(pC, MOVETO_ABS));

00057         pC = list_of(rx)(ry)(0)(0)(1)(x+w)(y+ry);
00058         pI.push_back( std::make_pair(pC, ARC_ABS));
00059         pC.clear();
00060         //perform a absolute vertical lineto to location (x+width,y+height-ry),
00061         pC = list_of(y+h-ry);
